    - brightnessThreshold=255: all cells should have brightness 0 (everything below threshold)
    - contrastBoost=0: reduced contrast (brightness values compressed toward center)
    - contrastBoost=200: increased contrast

    Invert charset:
    - invertCharset=true: character index is reversed

    Structural:
    - ImageData with very small dimensions (1x1) doesn't crash
    - ImageData with all-black pixels (fill 0) produces darkest characters
    - ImageData with all-white pixels (fill 255) produces brightest characters

function makeImageData(w: number, h: number, fill: number = 128) {
  const data = new Uint8ClampedArray(w * h * 4)
  // Fill RGBA channels — set RGB to fill, alpha to 255
  for (let i = 0; i < w * h * 4; i += 4) {
    data[i] = fill       // R
    data[i + 1] = fill   // G
    data[i + 2] = fill   // B
    data[i + 3] = 255    // A
  }
  return { width: w, height: h, data }
}

function makeGradientImage(w: number, h: number) {
  const data = new Uint8ClampedArray(w * h * 4)
  for (let y = 0; y < h; y++) {
    for (let x = 0; x < w; x++) {
      const i = (y * w + x) * 4
      const val = Math.floor((x / w) * 255)
