      <script type="module">
        // Generate synthetic test data inline
        function makeTestData() {
          const cols = 60, rows = 20, charset = ' .:-=+*#%@'
          const frames = []
          for (let f = 0; f < 30; f++) {
            const cells = []
            let text = ''
            for (let r = 0; r < rows; r++) {
              const row = []
              for (let c = 0; c < cols; c++) {
                const brightness = (Math.sin((c + f * 2) / 8) * Math.cos((r + f) / 6) + 1) / 2
                const charIdx = Math.floor(brightness * (charset.length - 1))
                const char = charset[charIdx]
                row.push({ char, r: 0, g: Math.floor(brightness * 255), b: 0, brightness })
                text += char
              }
              cells.push(row)
              text += '\n'
            }
            frames.push({ text, cells })
          }
          return {
            version: 1,
            metadata: { columns: cols, rows, fps: 12, duration: 30 / 12, frameCount: 30, colorMode: 'colored', charset },
            frames,
          }
        }

        const data = makeTestData()

        // Test 1: Web Component with .data property
        const el = document.getElementById('test1')
        el.data = data

        // Test 2: ES Module API
        import { AsciiPlayer } from './dist/index.js'
        const canvas = document.getElementById('test2')
        const player = new AsciiPlayer(canvas, data, {
          fps: 12, loop: 'forever', autoplay: true,
          fgColor: '#ffb000', bgColor: '#1a1200',
          font: '14px monospace',
        })
        player.ready.then(() => console.log('ES Module player ready'))
      </script>

      <p>Expected: Two animated ASCII art displays. Top uses Web Component with green-on-dark colors. Bottom uses ES Module API with amber colors.</p>
